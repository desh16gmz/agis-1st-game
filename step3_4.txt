using UnityEngine;

[RequireComponent(typeof(Rigidbody2D), typeof(BoxCollider2D))]
public class PlayerController : MonoBehaviour
{
    public float moveSpeed = 6f;
    public float jumpForce = 7f;

    private Rigidbody2D rb;
    private bool grounded;

    void Awake() => rb = GetComponent<Rigidbody2D>();

    void Update()
    {
        // Horizontal movement
        float h = Input.GetAxisRaw("Horizontal"); // A/D or ←/→
        rb.velocity = new Vector2(h * moveSpeed, rb.velocity.y);

        // Jump
        if (grounded && Input.GetButtonDown("Jump"))
        {
            rb.velocity = new Vector2(rb.velocity.x, jumpForce);
        }
    }

    void FixedUpdate()
    {
        grounded = IsGrounded();
    }

    // Simple ground check using a short downward raycast
    bool IsGrounded()
    {
        float rayLen = 0.1f;
        RaycastHit2D hit = Physics2D.BoxCast(
            transform.position,
            GetComponent<BoxCollider2D>().size * 0.9f,
            0f,
            Vector2.down,
            rayLen,
            LayerMask.GetMask("Block", "Default") // default for floor, Block for landed pieces
        );
        return hit.collider != null;
    }
}
